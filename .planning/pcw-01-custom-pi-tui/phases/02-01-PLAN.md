**Git Commands to Wrap:**
- `git status --porcelain` - Parse working tree status
- `git log --oneline -n 10` - Recent commits
- `git branch -a` - All branches
- `git checkout <branch>` - Switch branches
- `git checkout -b <branch>` - Create and switch to new branch
- `git branch -d <branch>` - Delete local branch
- `git add <file>` - Stage file
- `git reset HEAD <file>` - Unstage file
- `git rev-parse --git-dir` - Check if in git repo

## Tasks

### Task 1: Create git utility module

**Files:** `~/dev/pi-skills/extensions/dashboard/utils/git.ts`

**Interface:**
```typescript
export interface GitStatus {
  isRepo: boolean;
  branch: string;
  ahead: number;
  behind: number;
  modified: string[];
  staged: string[];
  untracked: string[];
}

export interface GitCommit {
  hash: string;
  message: string;
  author: string;
  date: string;
}

export async function getGitStatus(cwd: string): Promise<GitStatus>;
export async function getRecentCommits(cwd: string, count: number): Promise<GitCommit[]>;
export async function getBranches(cwd: string): Promise<{ name: string; current: boolean }[]>;
export async function checkoutBranch(cwd: string, branch: string): Promise<void>;
export async function createBranch(cwd: string, branch: string, checkout?: boolean): Promise<void>;
export async function deleteBranch(cwd: string, branch: string, force?: boolean): Promise<void>;
export async function stageFile(cwd: string, file: string): Promise<void>;
export async function unstageFile(cwd: string, file: string): Promise<void>;
```

**Verify:**
- Module exports all functions
- Functions have proper error handling
